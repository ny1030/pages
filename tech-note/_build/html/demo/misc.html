SQLチューニング¶

Level0. 実行計画(Query Plan)の読み方¶

#製品はPostgreSQLを例にしています

以下のようなSQLの実行計画を取得する。

UPDATE
      T_ITEM_INBOUND t3
    SET
      status_flag = '9'
      , updated_datetime = CURRENT_TIMESTAMP
      , updated_by = 'APP_001'
    FROM
      (
        SELECT
          t1.group_num
          , t1.level3_item_code
        FROM
          T_ITEM_INBOUND t1
        WHERE
          t1.status_flag = '1'
          AND t1.group_num = '10'
          AND t1.level3_item_code = '1000-2000-3000'
        GROUP BY
          t1.group_num
          , t1.level1_item_code
          , t1.level2_item_code
          , t1.level3_item_code
          , t1.color_code
          , t1.size_code
          , t1.pattern_length_code
        HAVING
          1   <   COUNT(t1.group_num)
      ) t2
    WHERE
      t3.status_flag = '1'
      AND t3.group_num = t2.group_num
      AND t3.level3_item_code = t2.level3_item_code
      AND t3.brand_code = 'TK'
      AND t3.region_code = 'JP'
